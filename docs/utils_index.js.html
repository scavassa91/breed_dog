/**
 * Split the breed to render and make requests
 * @param {String} key Dog breed split by '-'
 * @return {Object} Object with the breed and subBreed names
 */
export function splitBreed (key) {
  let breed;
  let subBreed;
  let split;

  if (key.indexOf('-')) {
    split = key.split('-');
    breed = split[0];
    subBreed = split[1];
  }
  return {
    breed,
    subBreed
  }
}
